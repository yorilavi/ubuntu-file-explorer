4. Create handler for move confirmation:
   ```typescript
   const handleMoveConfirm = useCallback(async (destDir: string) => {
     if (!moveTarget || !selectedServerId) return;

     const file = moveTarget;
     const sourceDir = file.path.substring(0, file.path.lastIndexOf('/')) || '/';

     setMoveTarget(null); // Close modal immediately

     try {
       const result = await window.electronAPI.moveFile(selectedServerId, file.path, destDir);

       if (result.success) {
         toast.success(`Moved "${file.name}" to ${destDir}`, {
           duration: 5000,
           action: {
             label: 'Undo',
             onClick: async () => {
               // Move back to original location
               const undoResult = await window.electronAPI.moveFile(
                 selectedServerId,
                 result.path!, // new path after move
                 sourceDir
               );
               if (undoResult.success) {
                 toast.success(`Moved "${file.name}" back`);
                 // Refresh current directory
                 refreshCurrentDirectory();
               } else {
                 toast.error(`Undo failed: ${undoResult.error}`);
               }
             },
           },
         });
         // Refresh current directory to show file is gone
         refreshCurrentDirectory();
       } else {
         toast.error(`Move failed: ${file.name}`, {
           description: result.error,
         });
       }
     } catch (error) {
       toast.error(`Move failed: ${file.name}`, {
         description: error instanceof Error ? error.message : 'Unknown error',
       });
     }
   }, [moveTarget, selectedServerId, refreshCurrentDirectory]);
   ```

5. Ensure refreshCurrentDirectory function exists or create it:
   - Should reload the current directory listing
   - Likely already exists for other file operations (check how delete refresh works)
