**8. Keyboard shortcuts for view mode -- add a new useEffect after the Cmd+Shift+. hidden files shortcut effect (around line 450):**
```typescript
useEffect(() => {
  const handleKeyDown = (e: KeyboardEvent) => {
    const target = e.target as HTMLElement;
    if (target.tagName === 'INPUT' || target.tagName === 'TEXTAREA' || target.isContentEditable) {
      return;
    }

    // Cmd+1 -> Column view, Cmd+2 -> List view (Finder convention)
    if (e.metaKey && !e.shiftKey && !e.altKey) {
      if (e.code === 'Digit1') {
        e.preventDefault();
        handleSetViewMode('columns');
      } else if (e.code === 'Digit2') {
        e.preventDefault();
        handleSetViewMode('list');
      }
    }
  };

  window.addEventListener('keydown', handleKeyDown);
  return () => window.removeEventListener('keydown', handleKeyDown);
}, [handleSetViewMode]);
```
