3. In src/preload/preload.ts:
   - Import CodeChunkData type (duplicate interface for preload isolation)
   - Add onCodeChunk method:
     ```typescript
     onCodeChunk: (callback: (data: CodeChunkData) => void): (() => void) => {
       const handler = (_event: unknown, data: CodeChunkData) => callback(data);
       ipcRenderer.on('preview:code-chunk', handler);
       return () => ipcRenderer.removeListener('preview:code-chunk', handler);
     },
     ```

     useEffect(() => {
       // Only stream if file might be large (we get streaming signal from first chunk)
       const unsubscribe = window.electronAPI.onCodeChunk((data) => {
         if (data.filePath === currentFilePath) {
           if (data.isInitial) {
             setIsStreaming(true);
             setStreamLines(data.chunk.split('\n'));
           } else {
             setStreamLines(prev => [...prev, ...data.chunk.split('\n')]);

           if (data.isComplete) {
             setStreamComplete(true);
           }
         }
       });
       return unsubscribe;
     }, [currentFilePath]);
