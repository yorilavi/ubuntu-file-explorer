    ```typescript
    // Track active operations by unique ID
    private activeOperations = new Map<string, AbortController>();

    // Generate unique operation ID
    private generateOperationId(): string {
      return `op-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
    }

    // Cancel an operation by ID
    public cancelOperation(operationId: string): boolean {
      const controller = this.activeOperations.get(operationId);
      if (controller) {
        controller.abort();
        this.activeOperations.delete(operationId);
        return true;
      }
      return false;
    }
    ```

    2. Update download method to accept and check abort signal:
    ```typescript
    async downloadFile(
      serverId: string,
      remotePath: string,
      localPath: string,
      operationId?: string,
      onProgress?: (progress: { bytes: number; total: number; percent: number }) => void
    ): Promise<{ operationId: string }> {
      const opId = operationId || this.generateOperationId();
      const controller = new AbortController();
      this.activeOperations.set(opId, controller);

      try {
        // Existing download logic but check controller.signal.aborted periodically
        // In SFTP fastGet callback, check if aborted
        const sftp = await this.getSftpClient(serverId);

        await new Promise<void>((resolve, reject) => {
          const readStream = sftp.createReadStream(remotePath);
          const writeStream = fs.createWriteStream(localPath);

          // Handle abort
          controller.signal.addEventListener('abort', () => {
            readStream.destroy();
            writeStream.destroy();
            reject(new Error('Operation cancelled'));
          });

          // Track progress
          let transferred = 0;
          readStream.on('data', (chunk) => {
            transferred += chunk.length;
            onProgress?.({ bytes: transferred, total: stats.size, percent: Math.round((transferred / stats.size) * 100) });
          });

          readStream.pipe(writeStream);
          writeStream.on('finish', resolve);
          writeStream.on('error', reject);
          readStream.on('error', reject);
        });

        return { operationId: opId };
      } finally {
        this.activeOperations.delete(opId);
      }
    }
    ```
