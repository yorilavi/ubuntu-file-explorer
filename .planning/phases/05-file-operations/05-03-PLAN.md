
2. Implement the function:
   ```typescript
   const refreshDirectory = useCallback(async (columnIndex: number) => {
     if (!connectedServerId) return;

     const column = columns[columnIndex];
     if (!column) return;

     try {
       const listing = await window.electronAPI.listDirectory(connectedServerId, column.path);

       // Update the column entries in state
       setColumns(prev => {
         const updated = [...prev];
         if (updated[columnIndex]) {
           updated[columnIndex] = {
             ...updated[columnIndex],
             entries: listing.entries.map(e => ({
               ...e,
               modified: new Date(e.modified), // IPC serialization fix
             })),
           };
         }
         return updated;
       });
     } catch (error) {
       console.error('[useColumnNavigation] Failed to refresh directory:', error);
     }
   }, [connectedServerId, columns]);
   ```

```typescript
const handleDownload = async () => {
  setContextMenu(null);
  const result = await window.electronAPI.downloadFile(serverId, entry.path, entry.name);
  if (result.success) {
    console.log(`Downloaded to: ${result.path}`);
  }
};

const handleUpload = async () => {
  setContextMenu(null);
  // Upload to current directory (entry is folder)
  const result = await window.electronAPI.uploadFile(serverId, entry.path);
  if (result.success) {
    onRefresh();
  }
};

const handleDelete = async () => {
  setContextMenu(null);
  const result = await window.electronAPI.deleteFile(
    serverId,
    entry.path,
    entry.name,
    entry.isDirectory
  );
  if (result.success) {
    onRefresh();
  }
};

const handleRenameStart = () => {
  setContextMenu(null);
  setRenameValue(entry.name);
  setIsRenaming(true);
};

const handleRenameConfirm = async () => {
  if (!renameValue.trim() || renameValue === entry.name) {
    setIsRenaming(false);
    return;
  }
  const result = await window.electronAPI.renameFile(serverId, entry.path, renameValue.trim());
  if (result.success) {
    onRefresh();
  }
  setIsRenaming(false);
};

const handleMoveTo = async () => {
  setContextMenu(null);
  // Use Electron's dialog to pick destination folder
  const result = await window.electronAPI.moveFileWithPicker(serverId, entry.path, entry.name);
  if (result.success) {
    onRefresh();
  }
};
```
